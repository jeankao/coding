from visual import*
from vphysics import *

G=6.673E-11
mass = {'earth': 5.97E24, 'moon': 7.36E22, 'sun':1.99E30}
radius = {'earth': 6.371E6*10, 'moon': 1.317E6*10, 'sun':6.95E8*10}
earth_orbit = {'r': 1.495E11, 'v': 2.9783E4}
moon_orbit = {'r': 3.84E8, 'v': 1.022E3}
theta = 20*pi/180.0

scene = display(width=800, height=1800, background=(0.5,0.5,0))

def G_force(s1, s2):
    r = s1.pos-s2.pos
    return - norm(r)*G*s1.m*s2.m/mag2(r)

earth = sphere(radius =radius['earth'], material = materials.earth)
moon = sphere(radius = radius['moon'])

earth.m, moon.m = mass['earth'], mass['moon']
emmass = earth.m + moon.m
re = moon_orbit['r'] * moon.m / emmass
rm = moon_orbit['r'] * earth.m / emmass
earth.pos = vector( - re*cos(theta), re*sin(theta), 0)
earth.v = vector(0, 0, moon_orbit['v']*(moon.m)/emmass)
#scene.forward = vector(0, -1, 0)
moon.pos = vector( rm*cos(theta), -rm*sin(theta), 0)
moon.v = vector(0, 0,  - moon_orbit['v']*(earth.m)/emmass)

dt= 60.0*60.0

while True:
    rate(1000)

    FGm = G_force(moon, earth)
    FGe = G_force(earth, moon)

    moon.a = FGm/moon.m
    moon.v += moon.a * dt
    moon.pos += moon.v *dt

    earth.a = FGe/earth.m
    earth.v += earth.a * dt
    earth.pos += earth.v * dt

    #scene.center = earth.pos
